void attack(char* aPassword)
{
  int j;
  for (j = 0 ; j < strlen(aPassword) ; j++) {
    typeLetter(aPassword[j]);
    delay(75);
  }
  typeEnter();
  delay(500);
}

void waitFunction()
{
  for (int k = 0; k < 6 ; k++)
  {
    typeEnter();
    delay(5000);
  }
  delay(150);
}
